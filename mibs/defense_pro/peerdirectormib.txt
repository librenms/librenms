    ACCESS not-accessible
    STATUS mandatory
    DESCRIPTION
      "Table containing the list of CIDRs."
    ::=  {rsCP 1}

rsCPCIDREntry OBJECT-TYPE
    SYNTAX RsCPCIDREntry
    ACCESS not-accessible
    STATUS mandatory
    DESCRIPTION
       " The row definition for this table. The table is made of four fields:
	 1) The CIDR's subnet.
	 2) The mask for the CIDR.
     3) The static assignment to peer IP.
     4) The static assignment to router IP.
	 The table's key is made of fields 1+2.
     Fields 3+4 are 0.0.0.0 by default and should always both be zero or a real IP."
    INDEX { rsCPCIDRSubnet,
            rsCPCIDRMask }
    ::=  {rsCPCIDRsTable 1}

  rsCPCIDRSubnet	                IpAddress,

-- Application-specific traps for PeerDirector
-- Each enterprise-specific trap has two bound variables describing the exact nature of
-- the trap. rndErrorDesc provides a detailed description of the problem, including the
-- related interface number, board number or any other applicable variable.
-- rndErrorSeverity describes the severity of this problem.

-- Trap template
-- -------------
--    TRAP-TYPE
--       ENTERPRISE  rsServerDispatcher
--       VARIABLES   { rndErrorDesc, rndErrorSeverity }
--       DESCRIPTION
--             ""
--       ::=
-- -------------

rsPDGeneralError TRAP-TYPE
         ENTERPRISE rsCP
         VARIABLES { rndErrorDesc, rndErrorSeverity }
         DESCRIPTION
             "This is a trap for all Peer Director's errors."
     ::=  1

rsPDInfoTrap TRAP-TYPE
         ENTERPRISE rsCP
         VARIABLES { rndErrorDesc, rndErrorSeverity }
     ::=  2

rsPDWarningTrap TRAP-TYPE
         ENTERPRISE rsCP
         VARIABLES { rndErrorDesc, rndErrorSeverity }
     ::=  3

rsPDErrorTrap TRAP-TYPE
         ENTERPRISE rsCP
         VARIABLES { rndErrorDesc, rndErrorSeverity }
     ::=  4
